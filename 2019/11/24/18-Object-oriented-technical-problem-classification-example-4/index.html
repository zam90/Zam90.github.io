31、利用面向对象的思想求解长方体体积

问题描述

有一个长方体类，其类名为box, 其私有数据成员包括length(长)、width(宽)、height(高)以及一个名为volume的公有成员函数，volume函数用来计算长方体的体积。现有3个长方体，分别为r1, r2, r3，3个长方体的长、宽、高(三者均为整数)从键盘分别输入，要求有序输出三个长方体的体积。

参考代码
#include <iostream>
#include <iomanip>
using namespace std;

class box
{
private:
    int length, width, height;

public:
    box(int a, int b, int c) : length(a), width(b), height(c)
    {
    }
    box() : length(0), width(0), height(0)
    {
    }
    int volume()
    {
        return height * length * width;
    }
};
int main()
{
    int a, b, c, d, e, f, g, h, i;
    cin >> a >> b >> c >> d >> e >> f >> g >> h >> i;
    box r1(a, b, c), r2(d, e, f), r3(g, h, i);
    box arr[3];
    arr[0] = r1;
    arr[1] = r2;
    arr[2] = r3;
    box *bp = arr;
    cout << bp[0].volume() << " ";
    cout << bp[1].volume() << " ";
    cout << bp[2].volume() << " ";
    bp = NULL;
    return 0;
}

32、构造函数与析构函数

问题描述

定义圆类Circle，利用类的复合（组合）定义圆柱类Column，然后创建圆柱类Column的实例化对象。

圆类Circle包含两个部分：

一、私有数据：点坐标x，y，半径r。

二、公有函数：所有函数要求类外定义

1、构造函数。

2、析构函数。

3、area函数：计算圆的面积。

4、print函数：输出点坐标x，y及半径r的值。

圆柱类Column包含两个部分：

一、私有数据：

1、类Circle的对象circle。

2、圆柱高height。

二、公有函数：所有函数要求类外定义

1、构造函数。

2、析构函数。

3、volume函数：计算圆柱的体积。要求使用circle.area（）。

4、print函数：输出点坐标x，y及半径r的值。

注：圆周率取3.14159

部分代码已在下面给出，要求不能改动已有代码

int main()
{
Column obj(2.3,3.4,4.5,5.6);
cout<<"The volume of obj is "<<obj.volume()<<endl;
return 0;
}

参考代码
#include <iostream>
using namespace std;
class Circle
{
public:
    Circle(double, double, double);
    ~Circle();
    float area();
    void print();

private:
    double x;
    double y;
    double r;
};

class Column
{
public:
    Column(double, double, double, double);
    ~Column();
    float volume();
    void print();

private:
    Circle circle;
    double height;
};

Circle::Circle(double a, double b, double c)
{
    x = a;
    y = b;
    r = c;
    cout << "Circle object start:"
         << "x=" << x << ",y=" << y << ",r=" << r << endl;
}

Circle::~Circle()
{
    cout << "Circle object end:x=" << x << ",y=" << y << ",r=" << r << endl;
}

float Circle::area()
{
    return (3.14159 * r * r);
}

void Circle::print()
{
    cout << "x=" << x << ",y=" << y << ",r=" << r << endl;
}

Column::Column(double h, double a, double b, double c) : circle(a, b, c)
{
    height = h;
    cout << "Column object start:height=" << height << ",";
    circle.print();
}

Column::~Column()
{
    cout << "Column object end:height=" << height << ",";
    circle.print();
}

float Column::volume()
{
    float m;
    m = circle.area();
    return (m * height);
}

int main()
{
    Column obj(2.3, 3.4, 4.5, 5.6);
    cout << "The volume of obj is " << obj.volume() << endl;
    return 0;
}

class Rectangle {
private:

    Rectangle ( double w = 0., double h = 0.) : width(w), height(h) { }

使得如下的函数能够正确地按照每个Rectangle对象的面积从小到大进行排序，并且main函数能通过编译且运行后能够得到样例输出。（请将下列代码复制到你补全后的Rectangle类之后，一并提交。） 

template<typename T>
void insertSort(T *ptr, int size) {
    for(int i = 1; i < size; ++i) {
        for( int j = i; j > 0; --j) {
            if( ptr[j] < ptr[j - 1]) {
                T temp = ptr[j];
                ptr[j ] = ptr[j - 1];
                ptr[j - 1] = temp;
            }
            else
                break;
        }
    }
}

int main () {

    data[0] = Rectangle(1,2);
    data[1] = Rectangle(5,7);

#include <iostream>
#include <iomanip>
using namespace std;

class Rectangle
{
private:
    double width, height;

public:
    Rectangle(double w = 0, double h = 0) : width(w), height(h) {}
    double getArea()
    {
        return width * height;
    }
    friend bool operator<(Rectangle a, Rectangle b)
    {
        if (a.getArea() > b.getArea())
            return 0;
        else
            return 1;
    }
};

template <typename T>
void insertSort(T *ptr, int size)
{
    for (int i = 1; i < size; ++i)
    {
        for (int j = i; j > 0; --j)
        {
            if (ptr[j] < ptr[j - 1])
            {
                T temp = ptr[j];
                ptr[j] = ptr[j - 1];
                ptr[j - 1] = temp;
            }
            else
                break;
        }
    }
}

int main()
{
    Rectangle data[4];
    data[0] = Rectangle(1, 2);
    data[1] = Rectangle(5, 7);
    data[2] = Rectangle(2, 8);
    data[3] = Rectangle(3, 4);
    for (int i = 0; i < 4; ++i)
    {
        printf("%f,", data[i].getArea());
    }
    printf("\n");
    insertSort(data, 4);
    for (int i = 0; i < 4; ++i)
    {
        printf("%f,", data[i].getArea());
    }
    return 0;
}

35、通过重载运算符来模拟时间

问题描述

有一个Time类，包含数据成员minute（分）和sec（秒），模拟秒表，每次走一秒，满60秒进一分钟，此时秒又从0起算，要求输出分和秒的值。

已有如下的main函数，要求编写能支持该main函数的Time类。

int main()
{
    Time time1(34,50);
    for(int i = 0; i < 20; i++)
    {
        ++time1;
        time1.display();
    }
    return 0;
}

参考代码
#include <iostream>
using namespace std;

class Time
{
private:
    int minute, sec;

public:
    Time(int a, int b)
    {
        if (b > 59)
        {
            minute = a + 1;
            sec = b - 60;
        }
        else
        {
            minute = a;
            sec = b;
        }
    }
    Time()
    {
        minute = 0;
        sec = 0;
    }
    void display()
    {
        cout << minute << ":" << sec << endl;
    }
    friend operator++(Time &a)
    {
        a.sec++;
        if (a.sec > 59)
        {
            a.minute = a.minute + 1;
            a.sec = a.sec - 60;
        }
    }
};

int main()
{
    Time time1(34, 50);
    for (int i = 0; i < 20; i++)
    {
        ++time1;
        time1.display();
    }
    return 0;
}
